choose: リストの先頭を選んで、その後の継続を取り、終わったら次々にリストの中身を渡す

let a = choose [1;2;3;4;5] in
let b = choose [1;2;3;4;5] in
let c = choose [1;2;3;4;5] in
if a*a + b*b = c*c then print_endline(a,b,c);
